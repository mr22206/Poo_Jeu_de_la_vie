    Standards de développement¶

    Style de code¶

    Nommage¶

    // Classes : PascalCase
class GameManager {
    // Méthodes : camelCase
    void updateGrid();

    // Variables membres : m_camelCase
    int m_gridSize;

    // Constantes : SCREAMING_SNAKE_CASE
    static const int MAX_GRID_SIZE = 1000;
};

// Énumérations : PascalCase
enum class CellState {
    Alive,
    Dead
};

    Formatage¶

    // Accolades : nouvelle ligne
void fonction()
{
    if (condition) {

    } else {

}

// Espacement
int fonction(int param1, int param2)
{
    int resultat = param1 + param2;
    return resultat;
}

    Tests¶

    Structure¶

    TEST_CASE("Nom du test", "[catégorie]") {
    // Arrangement
    GameManager game;

    SECTION("Description du scénario") {
        // Action
        bool result = game.update(0.16f);

        // Assertion
        REQUIRE(result == true);
    }
}

    Conventions¶

    1. Un fichier de test par classe
    2. Nommage explicite des cas de test
    3. Structure AAA (Arrange, Act, Assert)
    4. Tests des cas limites

    Validation des entrées¶

    void setGridSize(int size) {
    if (size <= 0 || size > MAX_GRID_SIZE) {
        throw std::invalid_argument("Taille de grille invalide");
    }
    m_gridSize = size;
}

    Gestion des ressources¶

    // RAII
class FileHandler {
    std::ifstream file;
public:
    explicit FileHandler(const std::string& path)
        : file(path) {
        if (!file) {
            throw std::runtime_error("Erreur d'ouverture");
        }
    }
    ~FileHandler() {
        if (file.is_open()) {
            file.close();
        }
    }
};
