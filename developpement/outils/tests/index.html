    Configuration des tests¶

    CMake¶

    # Tests configuration
enable_testing()
add_subdirectory(tests)

# Configuration de la couverture
if(CMAKE_BUILD_TYPE STREQUAL "Debug")
    include(CodeCoverage)
    append_coverage_compiler_flags()
    setup_target_for_coverage_lcov(
        NAME coverage
        EXECUTABLE ctest
        DEPENDENCIES unit_tests integration_tests
    )
endif()

    Outils d'analyse¶

    Valgrind¶

    #!/bin/bash

# Vérification des fuites mémoire
valgrind --leak-check=full \
         --show-leak-kinds=all \
         --track-origins=yes \
         --verbose \
         ./tests/unit_tests

# Profilage
valgrind --tool=callgrind ./tests/benchmarks

    Outils de couverture¶

    LCOV¶

    #!/bin/bash

# Génération rapport de couverture
lcov --capture --directory . \
     --output-file coverage.info

# Filtrage des fichiers système
lcov --remove coverage.info '/usr/*' --output-file coverage.info

# Génération HTML
genhtml coverage.info --output-directory coverage

    gcovr¶

    # Configuration XML pour Jenkins
gcovr --xml coverage.xml \
      --filter src/ \
      --exclude-unreachable-branches

    Outils de benchmark¶

    Google Benchmark¶

    #include <benchmark/benchmark.h>

static void BM_GridUpdate(benchmark::State& state) {
    Grid grid(state.range(0), state.range(0));
    for (auto _ : state) {
        grid.update();
    }
    state.SetComplexityN(state.range(0));
}

BENCHMARK(BM_GridUpdate)
    ->RangeMultiplier(2)
    ->Range(8, 1024)
    ->Complexity();

    GitHub Actions¶

    name: Tests

on: [push, pull_request]

jobs:
  test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v2

      - name: Install dependencies
        run: |
          sudo apt-get update
          sudo apt-get install -y cmake valgrind lcov

      - name: Build and test
        run: |
          mkdir build && cd build
          cmake -DCMAKE_BUILD_TYPE=Debug ..
          make
          ctest --output-on-failure

      - name: Generate coverage
        run: |
          make coverage
          bash <(curl -s https://codecov.io/bash)
