    CI/CD¶

    Pipeline GitHub Actions¶

    Workflow principal¶

    name: CI/CD Pipeline

on:
  push:
    branches: [ main, develop ]
  pull_request:
    branches: [ main, develop ]

jobs:
  build:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v2

      - name: Setup
        run: |
          sudo apt-get update
          sudo apt-get install -y cmake libsfml-dev

      - name: Build
        run: |
          mkdir build && cd build
          cmake ..
          make

      - name: Test
        run: |
          cd build
          ctest --output-on-failure

  analyze:
    needs: build
    runs-on: ubuntu-latest
    steps:
      - name: Static Analysis
        run: |
          cppcheck --enable=all src/

      - name: Security Scan
        uses: github/codeql-action/analyze@v1

  deploy:
    needs: [build, analyze]
    if: github.ref == 'refs/heads/main'
    runs-on: ubuntu-latest
    steps:
      - name: Create Release
        uses: actions/create-release@v1
        env:
          GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}

    Configuration Jenkins¶

    Jenkinsfile¶

    pipeline {
    agent any

    environment {
        BUILD_TYPE = 'Release'
    }

    stages {
        stage('Build') {
            steps {
                sh '''
                    mkdir -p build
                    cd build
                    cmake -DCMAKE_BUILD_TYPE=${BUILD_TYPE} ..
                    make -j$(nproc)
                '''
            }
        }

        stage('Test') {
            steps {
                sh '''
                    cd build
                    ctest --output-on-failure
                '''
            }
            post {
                always {
                    junit 'build/test-results/*.xml'
                }
            }
        }

        stage('Analysis') {
            parallel {
                stage('Static Analysis') {
                    steps {
                        sh 'cppcheck --xml --output-file=cppcheck-result.xml src/'
                    }
                }
                stage('Coverage') {
                    steps {
                        sh '''
                            cd build
                            make coverage
                        '''
                    }
                }
            }
        }

        stage('Deploy') {
            when {
                branch 'main'
            }
            steps {
                sh './scripts/deploy.sh'
            }
        }
    }

    post {
        always {
            cleanWs()
        }
    }
}

    Configuration GitLab CI¶

    .gitlab-ci.yml¶

    stages:
  - build
  - test
  - analyze
  - deploy

variables:
  BUILD_TYPE: Release

build:
  stage: build
  script:
    - mkdir -p build
    - cd build
    - cmake -DCMAKE_BUILD_TYPE=$BUILD_TYPE ..
    - make
  artifacts:
    paths:
      - build/

test:
  stage: test
  script:
    - cd build
    - ctest --output-on-failure
  artifacts:
    reports:
      junit: build/test-results/*.xml

analyze:
  stage: analyze
  script:
    - cppcheck --enable=all src/
    - cd build
    - make coverage
  artifacts:
    reports:
      coverage_report:
        coverage_format: cobertura
        path: coverage.xml

deploy:
  stage: deploy
  script:
    - ./scripts/deploy.sh
  only:
    - main

    Déploiement continu¶

    Script de déploiement¶

    #!/bin/bash

VERSION=$(git describe --tags)
DEPLOY_DIR="/var/www/jeu-de-vie"

# Préparation
mkdir -p dist
cp -r build/bin/* dist/
cp -r assets dist/
cp README.md LICENSE dist/

# Création archive
cd dist
tar czf jeu-de-vie-$VERSION.tar.gz *

# Déploiement
ssh deploy@server "mkdir -p $DEPLOY_DIR/$VERSION"
scp jeu-de-vie-$VERSION.tar.gz deploy@server:$DEPLOY_DIR/$VERSION/
ssh deploy@server "cd $DEPLOY_DIR/$VERSION && tar xzf jeu-de-vie-$VERSION.tar.gz"

# Mise à jour lien symbolique
ssh deploy@server "ln -sfn $DEPLOY_DIR/$VERSION $DEPLOY_DIR/current"
