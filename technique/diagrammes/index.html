        Obstacle

    class InputHandler {
        + ~InputHandler()

    class IEvolutionStrategy {
        + bool evolve(Cell* cell, int liveNeighbors) const = 0
        + string getName() const = 0
        + ~IEvolutionStrategy() = default
    }

    class ClassicEvolution {
        - string name = "Classic"
        + ClassicEvolution()
        + bool evolve(Cell* cell, int liveNeighbors) const override
        + string getName() const override
        + ~ClassicEvolution() override
    }

    class HighLifeEvolution {
        - string name = "High Life"
        + HighLifeEvolution()
        + bool evolve(Cell* cell, int liveNeighbors) const override
        + string getName() const override
        + ~HighLifeEvolution() override
    }

    class Subject {
        - std::vector<std::shared_ptr<Observer>> observers
        + void attach(std::shared_ptr<Observer> observer)
        + void detach(std::shared_ptr<Observer> observer)
        + void notify()
        + ~Subject()
    }

    class Observer {
        + void update(Subject* subject) = 0
        + ~Observer()
    }

    class IRenderer {
        + void render(Grid* grid)
        + string getName() const

        + ~IRenderer()
    }

        + string getName() const override
        + ~ConsoleRenderer() override
    }

    class GraphicRenderer {

        + void render(Grid* grid) override

        + string getName() const override
        + ~GraphicRenderer() override
    }

        + ~FileHandler()
    }

### Classes principales
- **Grid** : Gestion de la grille de jeu
- **Cell** : Représentation d'une cellule
- **Game** : Moteur du jeu
- **Renderer** : Interface d'affichage
  - ConsoleRenderer
  - SFMLRenderer

### Flux principal
1. Initialisation du jeu
2. Boucle de simulation
3. Mise à jour de l'affichage
4. Gestion des événements

    Hiérarchie des classes

    • Game : Contrôleur principal
    • Grid : Gestion de l'état
    • Cell : Unité de base

    Flux de données

    1. Input utilisateur → Game
    2. Game → Grid (évolution)
    3. Grid → Renderer (affichage)
    4. Renderer → Utilisateur (feedback)
