    Observer¶

    // Subject
class Grid : public Subject {
    void notifyObservers() override {
        for(auto observer : observers) {
            observer->update();
        }
    }
};

// Observer
class IRenderer : public Observer {
    virtual void update() = 0;
};

    Factory¶

    class RendererFactory {
public:
    static unique_ptr<IRenderer> createRenderer(RendererType type) {
        switch(type) {
            case RendererType::Console:
                return make_unique<ConsoleRenderer>();
            case RendererType::Graphic:
                return make_unique<GraphicRenderer>();
        }
    }
};

    Singleton¶

    class ConfigManager {
private:
    static ConfigManager* instance;
    ConfigManager() {}

public:
    static ConfigManager* getInstance() {
        if(!instance) {
            instance = new ConfigManager();
        }
        return instance;
    }
};

    Command¶

    class Command {
public:
    virtual void execute() = 0;
    virtual void undo() = 0;
};

class ToggleCellCommand : public Command {
    Grid& grid;
    int x, y;
public:
    void execute() override;
    void undo() override;
};
