    Tests Unitaires

    Grid

    class GridTest : public ::testing::Test {
protected:
    void SetUp() override {
        grid = std::make_unique<Grid>(10, 10);
    }

    std::unique_ptr<Grid> grid;
};

TEST_F(GridTest, InitializationTest) {
    EXPECT_EQ(grid->getWidth(), 10);
    EXPECT_EQ(grid->getHeight(), 10);
    EXPECT_FALSE(grid->getCell(5, 5));
}

TEST_F(GridTest, CellManipulationTest) {
    grid->setCell(1, 1, true);
    EXPECT_TRUE(grid->getCell(1, 1));

    grid->setCell(1, 1, false);
    EXPECT_FALSE(grid->getCell(1, 1));
}

    Rules

    TEST(RulesTest, BasicEvolution) {
    Rules rules;
    Cell cell;

    // Test naissance
    EXPECT_TRUE(rules.computeNextState(cell, 3)) 
        << "Une cellule morte avec 3 voisins devrait naître";

    // Test survie
    cell.setAlive(true);
    EXPECT_TRUE(rules.computeNextState(cell, 2))
        << "Une cellule vivante avec 2 voisins devrait survivre";

    // Test mort par solitude
    EXPECT_FALSE(rules.computeNextState(cell, 1))
        << "Une cellule vivante avec 1 voisin devrait mourir";
}

    Tests d'Intégration

    Game Loop

    TEST(GameIntegrationTest, FullCycle) {
    Game game(800, 600);

    // Configuration initiale
    ASSERT_TRUE(game.initialize());

    // Simulation d'un cycle complet
    game.update();
    EXPECT_NO_THROW(game.render());

    // Vérification de l'état
    const auto& grid = game.getGrid();
    EXPECT_GT(grid.getAliveCells(), 0);
}

    Persistance

    TEST(PersistenceTest, SaveLoadState) {
    Game game;
    game.initialize();

    // Sauvegarde
    ASSERT_TRUE(game.saveState("test.save"));

    // Modification
    game.reset();

    // Chargement
    ASSERT_TRUE(game.loadState("test.save"));

    Tests de Performance

    Benchmarks

    TEST(PerformanceTest, GridEvolution) {
    Grid grid(100, 100);

    auto start = std::chrono::high_resolution_clock::now();
    for(int i = 0; i < 1000; i++) {
        grid.evolve();
    }
    auto end = std::chrono::high_resolution_clock::now();

    auto duration = std::chrono::duration_cast<std::chrono::milliseconds>
        (end - start).count();

    EXPECT_LT(duration, 1000) 
        << "1000 évolutions devraient prendre moins d'une seconde";
}

    Exécution des Tests

    Commandes

    # Compilation et exécution
cmake -DBUILD_TESTING=ON ..
make
ctest --output-on-failure

# Tests spécifiques
./tests/unit_tests --gtest_filter=GridTest.*
./tests/unit_tests --gtest_filter=RulesTest.*

    Options de test

    Option Description
    --gtest_filter Filtre les tests à exécuter
    --gtest_repeat Répète les tests N fois
    --gtest_break_on_failure Arrête à la première erreur
    --gtest_output Format de sortie (XML, JSON)
